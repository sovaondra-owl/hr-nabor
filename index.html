1. Příprava Google Sheetu

A

Otevřete svůj Google Sheet a v menu zvolte Rozšíření → Apps Script.

B

Smažte veškerý obsah editoru a vložte tento kód:

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  var mode = data.mode || "replace";
  var headers = data.headers || [];
  var rows = data.rows || [];

  if (rows.length === 0 && mode !== "update") {
    return ContentService.createTextOutput(JSON.stringify({ status: "ok", count: 0 })).setMimeType(ContentService.MimeType.JSON);
  }

  if (mode === "replace") {
    if (headers.length > 0) {
      sheet.clear();
      sheet.appendRow(headers);
    }
    rows.forEach(function(row) { sheet.appendRow(row); });
    return ContentService.createTextOutput(JSON.stringify({ status: "ok", count: rows.length })).setMimeType(ContentService.MimeType.JSON);
  }

  if (mode === "update") {
    var lastRow = sheet.getLastRow();
    var idCol = 1;
    var existingById = {};
    if (lastRow >= 2) {
      var idRange = sheet.getRange(2, idCol, lastRow, idCol);
      var ids = idRange.getValues();
      for (var i = 0; i < ids.length; i++) { existingById[String(ids[i][0])] = i + 2; }
    } else {
      sheet.clear();
      if (headers.length > 0) sheet.appendRow(headers);
    }
    var updated = 0, appended = 0;
    rows.forEach(function(row) {
      var id = row[0] != null ? String(row[0]) : "";
      if (existingById[id]) {
        var r = existingById[id];
        for (var c = 0; c < row.length; c++) sheet.getRange(r, c + 1).setValue(row[c] != null ? row[c] : "");
        updated++;
      } else {
        sheet.appendRow(row);
        appended++;
      }
    });
    return ContentService.createTextOutput(JSON.stringify({ status: "ok", updated: updated, appended: appended })).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ status: "ok", count: rows.length })).setMimeType(ContentService.MimeType.JSON);
}
